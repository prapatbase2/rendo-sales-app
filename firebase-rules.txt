rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    function signedIn() { return request.auth != null; }
    function mePath() { return /databases/$(database)/documents/users/$(request.auth.uid); }
    function hasMe() { return signedIn() && exists(mePath()); }
    function me() { return get(mePath()).data; }
    function active() { return hasMe() && me().active == true; }
    function role() { return me().role; }
    function isOwner() { return active() && role() == 'owner'; }
    function isManager() { return active() && role() == 'manager'; }
    function isSupervisor() { return active() && role() == 'supervisor'; }
    function isOwnerManager() { return isOwner() || isManager(); }
    function isAdminAttendance() { return isOwnerManager() || isSupervisor(); }
    function isLowerRoleName(r) {
      return r in ['front_kitchen','back_kitchen','front_staff','rotating_staff','daily'];
    }
    function isLowerUser() { return active() && isLowerRoleName(role()); }
    function isSalesRole() { return active() && role() in ['owner','manager','supervisor','front_staff']; }
    function currentMonthOrdinal() { return request.time.year() * 12 + request.time.month(); }
    function recentMonth(data) {
      return data.monthOrdinal is int
        && data.monthOrdinal >= currentMonthOrdinal() - 1
        && data.monthOrdinal <= currentMonthOrdinal();
    }
    function canCreateRole(targetRole) {
      return (isOwner())
        || (isManager() && targetRole != 'owner')
        || (isSupervisor() && isLowerRoleName(targetRole));
    }
    function firstOwnerUserCreate(uid) {
      return signedIn()
        && uid == request.auth.uid
        && request.resource.data.role == 'owner'
        && !exists(/databases/$(database)/documents/system/initialization)
        && getAfter(/databases/$(database)/documents/system/initialization).data.ownerUid == uid;
    }

    match /system/initialization {
      allow read: if true;
      allow create: if signedIn()
        && !exists(/databases/$(database)/documents/system/initialization)
        && request.resource.data.initialized == true
        && request.resource.data.ownerUid == request.auth.uid;
      allow update: if isOwner();
      allow delete: if false;
    }

    match /system/backupQueue {
      allow read: if isOwner();
      allow create, update: if active();
      allow delete: if false;
    }

    match /system/{other} {
      allow read, write: if false;
    }

    match /users/{uid} {
      allow read: if signedIn() && (active() || request.auth.uid == uid);
      allow create: if firstOwnerUserCreate(uid)
        || (active() && canCreateRole(request.resource.data.role)
          && request.resource.data.active == true
          && request.resource.data.loginId is string);
      allow update: if active() && (
        (isOwner() && uid != request.auth.uid)
        || (isOwner() && uid == request.auth.uid
          && request.resource.data.role == resource.data.role
          && request.resource.data.active == resource.data.active
          && request.resource.data.loginId == resource.data.loginId)
        || (isManager() && uid != request.auth.uid
          && !(resource.data.role in ['owner','manager'])
          && request.resource.data.role != 'owner')
      );
      allow delete: if false;
    }

    match /userPins/{uid} {
      allow read: if isOwner() || (active() && request.auth.uid == uid);
      allow create: if firstOwnerUserCreate(uid)
        || (active() && canCreateRole(getAfter(/databases/$(database)/documents/users/$(uid)).data.role));
      allow update: if isOwner() || (active() && request.auth.uid == uid);
      allow delete: if false;
    }

    match /securityKeys/pinPublic {
      allow read: if active();
      allow create: if signedIn()
        && !exists(/databases/$(database)/documents/system/initialization)
        && getAfter(/databases/$(database)/documents/system/initialization).data.ownerUid == request.auth.uid;
      allow update: if isOwner();
      allow delete: if false;
    }

    match /ownerSecrets/{docId} {
      allow read, update: if isOwner();
      allow create: if signedIn()
        && !exists(/databases/$(database)/documents/system/initialization)
        && getAfter(/databases/$(database)/documents/system/initialization).data.ownerUid == request.auth.uid;
      allow delete: if false;
    }

    match /appSettings/main {
      allow read: if active();
      allow create: if signedIn()
        && !exists(/databases/$(database)/documents/system/initialization)
        && getAfter(/databases/$(database)/documents/system/initialization).data.ownerUid == request.auth.uid;
      allow update: if isOwner();
      allow delete: if false;
    }

    match /privateSettings/{docId} {
      allow read, create, update: if isOwner();
      allow delete: if false;
    }

    match /payrollProfiles/{uid} {
      allow read, create, update: if isOwnerManager();
      allow delete: if false;
    }

    match /attendance/{attendanceId} {
      allow read: if active() && (isAdminAttendance() || (isLowerUser() && (resource.data.userId == request.auth.uid || recentMonth(resource.data))));
      allow create: if isLowerUser()
        && request.resource.data.userId == request.auth.uid
        && request.resource.data.role == role();
      allow update: if active() && (
        (isLowerUser()
          && resource.data.userId == request.auth.uid
          && request.resource.data.userId == request.auth.uid
          && request.resource.data.role == resource.data.role)
        || (isOwnerManager()
          && request.resource.data.diff(resource.data).affectedKeys().hasOnly(['paid','updatedAt','updatedBy']))
      );
      allow delete: if false;
    }

    match /dailySales/{dateId} {
      allow read: if active() && (
        role() in ['owner','manager','supervisor']
        || (isLowerUser() && recentMonth(resource.data))
      );
      allow create, update: if isSalesRole()
        && request.resource.data.date == dateId
        && request.resource.data.status == 'final';
      allow delete: if false;
    }

    match /dailySalesDrafts/{dateId} {
      allow read, create, update, delete: if isSalesRole();
    }

    match /salaryAdvances/{advanceId} {
      allow read: if isOwnerManager()
        || (active() && resource.data.userId == request.auth.uid && me().canUseAdvance == true);
      allow create: if isOwnerManager()
        || (active() && request.resource.data.userId == request.auth.uid && me().canUseAdvance == true);
      allow update: if isOwnerManager()
        || (active()
          && resource.data.userId == request.auth.uid
          && request.resource.data.userId == request.auth.uid
          && me().canUseAdvance == true);
      allow delete: if false;
    }

    match /compensationMonthSettings/{month} {
      allow read, create: if isOwnerManager();
      allow update: if isOwnerManager()
        && (!(resource.data.status == 'finalized' && request.resource.data.status == 'draft') || isOwner());
      allow delete: if false;
    }

    match /compensationRecords/{recordId} {
      allow read, create, update: if isOwnerManager();
      allow delete: if false;
    }

    match /recurringExpenses/{id} {
      allow read, create, update: if isOwnerManager();
      allow delete: if false;
    }
    match /recurringExpenseSnapshots/{id} {
      allow read, create, update: if isOwnerManager();
      allow delete: if false;
    }
    match /ownerExpenses/{id} {
      allow read, create, update: if isOwnerManager();
      allow delete: if false;
    }

    match /auditLogs/{logId} {
      allow read: if isOwner() || (isManager() && resource.data.hidden == false);
      allow create: if (active()
          && request.resource.data.actorId == request.auth.uid
          && request.resource.data.hidden == false)
        || (!exists(/databases/$(database)/documents/system/initialization)
          && request.resource.data.actorId == request.auth.uid);
      allow update: if isOwner()
        && request.resource.data.diff(resource.data).affectedKeys().hasOnly(['hidden','hiddenAt','hiddenBy']);
      allow delete: if false;
    }

    match /backupsMetadata/{id} {
      allow read, create: if isOwner();
      allow update, delete: if false;
    }

    match /{document=**} {
      allow read, write: if false;
    }
  }
}
